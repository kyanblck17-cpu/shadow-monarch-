<script>
const COLORS={mon:'#2E75B6',tue:'#C0392B',wed:'#27AE60',thu:'#8E44AD',fri:'#D35400',sat:'#16A085',sun:'#C9A84C'};

function showDay(day){
  document.querySelectorAll('.day-view').forEach(v=>v.classList.remove('active'));
  document.querySelectorAll('.day-btn').forEach(b=>b.classList.remove('active'));
  const view=document.getElementById('day-'+day);
  const btn=document.querySelector('[data-day="'+day+'"]');
  if(view)view.classList.add('active');
  if(btn){
    btn.classList.add('active');
    btn.style.borderBottomColor=COLORS[day];
    btn.querySelector('.d-lbl').style.color=COLORS[day];
  }
  // Reset others
  document.querySelectorAll('.day-btn:not(.active)').forEach(b=>{
    b.style.borderBottomColor='transparent';
    b.querySelector('.d-lbl').style.color='';
  });
  // Scroll to top
  window.scrollTo({top:0,behavior:'smooth'});
}

// Apply left border colours to all rows
document.querySelectorAll('[id^="day-"]').forEach(view=>{
  const day=view.id.replace('day-','');
  const color=COLORS[day];
  if(!color)return;
  view.querySelectorAll('.sched-row').forEach(r=>{
    r.style.setProperty('--row-color',color);
    // Apply left border via pseudo handled in CSS — use box-shadow instead
    r.style.boxShadow=`inset 2px 0 0 ${color}44`;
    r.style.borderLeftColor=color+'33';
  });
  view.querySelectorAll('.sched-row.highlight').forEach(r=>{
    r.style.boxShadow=`inset 3px 0 0 ${color}`;
  });
});

// Init
showDay('mon');
</script>
